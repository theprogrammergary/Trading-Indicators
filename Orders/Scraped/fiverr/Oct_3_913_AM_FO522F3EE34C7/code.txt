indicator("for_millerboy123", overlay = true)

buy_signal = ta.crossover(open, money_line)

plotshape(sell_signal, title = 'SELL', style = shape.labeldown, location = location.abovebar, 
 color = color.red, textcolor = color.white, size = size.small, text = 'SELL')
